<script>
// kattintásra megjelenő szöveg
function showMessage(e) {
    const msg = document.getElementById('message');
    msg.style.opacity = '1';
    createMorsha(e.clientX, e.clientY);
}

// apró morzsák létrehozása
function createMorsha(x, y) {
    for (let i = 0; i < 5; i++) {
        const morsha = document.createElement('div');
        morsha.className = 'morsha';
        document.body.appendChild(morsha);

        morsha.style.left = x + 'px';
        morsha.style.top = y + 'px';
        morsha.style.opacity = '1';

        const angle = Math.random() * 2 * Math.PI;
        const distance = 30 + Math.random() * 20;
        const dx = Math.cos(angle) * distance;
        const dy = Math.sin(angle) * distance;

        morsha.animate([
            { transform: 'translate(0px,0px)', opacity: 1 },
            { transform: `translate(${dx}px, ${dy}px)`, opacity: 0 }
        ], {
            duration: 1500 + Math.random() * 500,
            easing: 'ease-out'
        });

        setTimeout(() => {
            morsha.remove();
        }, 2000);
    }
}
</script>
